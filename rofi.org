#+TITLE: ROFI
#+setupfile: setup.org


We open the configuration section as follows:


* Configuration
#+BEGIN_SRC text :tangle ~/.config/rofi/rofi.rasi :results output silent
configuration {
#+END_SRC


** Core parameters
#+BEGIN_SRC text :tangle ~/.config/rofi/rofi.rasi :results output silent
modi: "window,run,drun,filebrowser";
matching: "fuzzy";
terminal: "kitty";
#+END_SRC

** Appearance
#+BEGIN_SRC text :tangle ~/.config/rofi/rofi.rasi :results output silent

#+END_SRC

** Keyboard shortcuts
These are intended to mimick my emacs setup. *the* killer feature of rofi!
*** Moving around
#+BEGIN_SRC text :tangle ~/.config/rofi/rofi.rasi :results output silent
kb-row-down: "Down,Alt+k";
kb-row-up: "Up,Alt+i";
kb-move-char-back: "Left,Alt+j";
kb-move-char-forward: "Right,Alt+l";
kb-move-word-back: "Alt+u,Control+Left";
kb-move-word-forward: "Alt+o,Control+Right";
kb-move-front: "Alt+h";
kb-move-end: "Alt+m";      
#+END_SRC

*** Deleting
#+BEGIN_SRC text :tangle ~/.config/rofi/rofi.rasi :results output silent
kb-remove-char-back: "BackSpace,Alt+d";
kb-remove-char-forward: "Delete,Alt+f";
kb-remove-word-back: "Alt+e,Control+BackSpace";
kb-remove-word-forward: "Alt+r";        
kb-remove-to-eol: "Control+k";
#+END_SRC
** filebrowser configuration
=rofi= refuses to take ="~"= as a default directory, but using ="."= works just the same for some reason.

#+BEGIN_SRC text :tangle ~/.config/rofi/rofi.rasi :results output silent
filebrowser {
/** Directory the file browser starts in. */
directory: ".";
/**
  ,* Sorting method. Can be set to:
  ,*   - "name"
  ,*   - "mtime" (modification time)
  ,*   - "atime" (access time)
  ,*   - "ctime" (change time)
  ,*/
sorting-method: "mtime";
/** Group directories before files. */
directories-first: true;
}
#+END_SRC
** Finalization
Closing the =configuration= section
#+BEGIN_SRC text :tangle ~/.config/rofi/rofi.rasi :results output silent
}
#+END_SRC
