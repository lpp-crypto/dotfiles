#+TITLE: ROFI
#+setupfile: setup.org


We open the configuration section as follows:


* Configuration
** Core parameters
#+NAME: rofi-config
#+BEGIN_SRC css :noweb yes
modi: "window,run,drun";
matching: "fuzzy";
terminal: "kitty";
#+END_SRC

** Appearance
#+NAME: rofi-appearance
#+BEGIN_SRC css :noweb yes
width: 50;
lines: 15;
columns: 1;
bw: 1;
location: 0;
line-margin: 3;
line-padding: 5;
padding: 20;
yoffset: 0;
xoffset: 0;
fixed-num-lines: true;
font: "Iosevka SS01 Medium 14";
show-icons: true;
icon-theme: "Zafiro-Nord-Black-Blue";
#+END_SRC

** Keyboard shortcuts
These are intended to mimick my emacs setup. *the* killer feature of rofi!
*** Moving around
#+NAME: rofi-kbd-moving
#+BEGIN_SRC css :noweb yes
kb-row-down: "Down,Alt+k";
kb-row-up: "Up,Alt+i";
kb-move-char-back: "Left,Alt+j";
kb-move-char-forward: "Right,Alt+l";
kb-move-word-back: "Alt+u,Control+Left";
kb-move-word-forward: "Alt+o,Control+Right";
kb-move-front: "Alt+h";
kb-move-end: "Alt+m";      
#+END_SRC

*** Deleting
#+NAME: rofi-kbd-deleting
#+BEGIN_SRC css :noweb yes
kb-remove-char-back: "BackSpace,Alt+d";
kb-remove-char-forward: "Delete,Alt+f";
kb-remove-word-back: "Alt+e,Control+BackSpace";
kb-remove-word-forward: "Alt+r";        
kb-remove-to-eol: "Control+k";
#+END_SRC
** Finalization
We write the above content to each file.

#+NAME: rofi-setup
#+BEGIN_SRC :css :noweb yes
  configuration {
  <<rofi-config>>
  <<rofi-appearance>>
  <<rofi-kbd-moving>>
  <<rofi-kbd-deleting>>
  }
#+END_SRC

** dmenu
The number of lines must be even or else the colors form a checkboard pattern instead of rows.

#+BEGIN_SRC css :tangle ~/.config/rofi/dmenu.rasi :results output silent :noweb yes
  <<rofi-setup>>

  <<rofi-theme-common>>

  dynamiccontent {
      orientation: vertical;
      children: [listview];
  }

  listview {
      columns: 6;
      lines: 3;
      fixed-height: false;
      layout: vertical;
      flow: horizontal;
      spacing: 4px;
  }
#+END_SRC
